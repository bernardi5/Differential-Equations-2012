% here is the macro \caution: it prints a marginal note containing the text in
% braces. Note that if the variable \madechanges is set to zero, every \caution
% command sequence is ignored.  To hide the marginal notes just comment the
% \madechanges=1 line and uncomment the \madechanges=0.
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
\newcount\madechanges \madechanges=1 \oddsidemargin=16pt \marginparwidth=3.4cm
%\madechanges=0
\def\caution#1{\ifnum \madechanges=1 \affixmessage{#1}% 
\else \relax \fi}
\def\affixmessage#1{\marginpar{{\footnotesize  \em #1} \openup
-.3\baselineskip}}
